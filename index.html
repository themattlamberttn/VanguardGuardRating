// VANGUARD SECURITY — Guard Ratings v4 function doGet(e) { try { const action = e.parameter.action || 'get'; const quarter = e.parameter.quarter || 'Q1 2026'; if (action === 'get') { const sheet = getSheet(quarter); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return respond({ guards: [] }); const guards = []; for (let i = 1; i < data.length; i++) { const r = data[i]; if (!r[0]) continue; guards.push({ name: r[0], role: r[1], ratings: { punctuality: Number(r[2]) || 0, professionalism: Number(r[3]) || 0, reliability: Number(r[4]) || 0, workEthic: Number(r[5]) || 0 }, teamPerf: Number(r[6]) || 0, notes: r[7] || '', coaching: safeJson(r[8]) }); } return respond({ guards }); } if (action === 'init') { const sheet = getSheet(quarter); const guardsParam = e.parameter.guards || '[]'; const incoming = safeJson(guardsParam); const data = sheet.getDataRange().getValues(); const existing = new Set(data.slice(1).map(r => r[0])); const toAdd = incoming.filter(g => !existing.has(g.name)); toAdd.forEach(g => sheet.appendRow([g.name, g.role, '', '', '', '', '', '', '[]'])); return respond({ ok: true, added: toAdd.length }); } if (action === 'save') { const sheet = getSheet(quarter); const g = safeJson(e.parameter.guard); if (!g || !g.name) return respond({ error: 'No guard data' }); // Calculate overall and flag const ratings = [g.ratings.punctuality, g.ratings.professionalism, g.ratings.reliability, g.ratings.workEthic].filter(v => v > 0); const overall = ratings.length ? Math.round((ratings.reduce((a,b) => a+b, 0) / ratings.length) * 10) / 10 : 0; const flag = overall > 0 ? (overall <= 2.0 ? 'Q' : 'NQ') : ''; const row = [ g.name, g.role, g.ratings.punctuality || '', g.ratings.professionalism || '', g.ratings.reliability || '', g.ratings.workEthic || '', g.teamPerf || '', g.notes || '', JSON.stringify(g.coaching || []), overall || '', flag ]; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === g.name) { sheet.getRange(i + 1, 1, 1, 11).setValues([row]); // Color the flag cell colorFlagCell(sheet, i + 1, 11, flag); updateSupervisorTab(g, overall, flag, quarter); return respond({ ok: true }); } } sheet.appendRow(row); const lastRow = sheet.getLastRow(); colorFlagCell(sheet, lastRow, 11, flag); updateSupervisorTab(g, overall, flag, quarter); return respond({ ok: true }); } if (action === 'delete') { const sheet = getSheet(quarter); const name = e.parameter.name; const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === name) { sheet.deleteRow(i + 1); break; } } return respond({ ok: true }); } if (action === 'clearAll') { const sheet = getSheet(quarter); const last = sheet.getLastRow(); if (last > 1) sheet.getRange(2, 3, last - 1, 9).clearContent(); return respond({ ok: true }); } return respond({ error: 'Unknown action' }); } catch(err) { return respond({ error: err.toString() }); } } // ─── SHEET HELPERS ──────────────────────────────────────────────────────────── function getSheet(quarter) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const name = (quarter || 'Q1 2026').replace(/[^a-zA-Z0-9 ]/g, '').trim().slice(0, 30); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); const headers = ['Name','Role','Punctuality','Professionalism','Reliability','Work Ethic','Team Perf','Notes','Coaching Log','Overall','Flag']; sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold').setBackground('#003087').setFontColor('#ffffff'); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 200); sheet.setColumnWidth(8, 200); sheet.setColumnWidth(9, 300); sheet.setColumnWidth(10, 70); sheet.setColumnWidth(11, 60); } return sheet; } function colorFlagCell(sheet, row, col, flag) { const cell = sheet.getRange(row, col); if (flag === 'Q') { cell.setBackground('#E1F5EE').setFontColor('#1D9E75').setFontWeight('bold'); } else if (flag === 'NQ') { cell.setBackground('#FCEBEB').setFontColor('#A32D2D').setFontWeight('bold'); } else { cell.setBackground('#ffffff').setFontColor('#6b6b6b').setFontWeight('normal'); } } // ─── SUPERVISOR TAB ─────────────────────────────────────────────────────────── function updateSupervisorTab(g, overall, flag, quarter) { if (g.role !== 'Supervisor') return; const ss = SpreadsheetApp.getActiveSpreadsheet(); const tabName = 'Supervisor Performance'; let sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); const headers = ['Name','Quarter','Punctuality','Professionalism','Reliability','Work Ethic','Team Performance','Overall','Flag','Notes']; sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold').setBackground('#003087').setFontColor('#ffffff'); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 200); sheet.setColumnWidth(10, 220); } const row = [ g.name, quarter, g.ratings.punctuality || '', g.ratings.professionalism || '', g.ratings.reliability || '', g.ratings.workEthic || '', g.teamPerf || '', overall || '', flag, g.notes || '' ]; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === g.name && data[i][1] === quarter) { sheet.getRange(i + 1, 1, 1, row.length).setValues([row]); colorFlagCell(sheet, i + 1, 9, flag); return; } } sheet.appendRow(row); const lastRow = sheet.getLastRow(); colorFlagCell(sheet, lastRow, 9, flag); } // ─── UTILS ──────────────────────────────────────────────────────────────────── function safeJson(val) { try { return JSON.parse(val || '[]'); } catch(e) { return []; } } function respond(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
